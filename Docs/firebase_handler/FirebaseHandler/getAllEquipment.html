Set<String> getAllEquipment() {
  var allEquipment = <String>{};
  for (var room in getAllRooms().values) {
    for (var workspace in room.workspaces.values) {
      for (var equipment in workspace) {
        allEquipment.add(equipment);
      }
    }
  }
  allEquipment.remove("");
  return allEquipment;
}
